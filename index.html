export default { async fetch(request, env) { const corsHeaders = { "Access-Control-Allow-Origin": "https://vasco-sketch.github.io", "Access-Control-Allow-Methods": "POST, OPTIONS, GET", "Access-Control-Allow-Headers": "Content-Type" }; if (request.method === "OPTIONS") { return new Response(null, { status: 204, headers: corsHeaders }); } if (request.method === "GET") { return Response.json( { estado: "Worker ligado", mensagem: "O Worker do quiz com IA está a funcionar." }, { headers: corsHeaders } ); } try { const body = await request.json(); const pergunta = body.pergunta || ""; const respostaCerta = body.respostaCerta || ""; const respostaAluno = body.respostaAluno || ""; if (!env.GEMINI_API_KEY) { return Response.json( { certa: false, feedback: "A chave GEMINI_API_KEY não está configurada no Worker." }, { status: 500, headers: corsHeaders } ); } if (!pergunta || !respostaCerta || !respostaAluno) { return Response.json( { certa: false, feedback: "Faltam dados para corrigir a resposta." }, { status: 400, headers: corsHeaders } ); } const prompt = ` És um corretor de quiz em português de Portugal. Avalia se a resposta do aluno está correta. Regras: - Aceita respostas com palavras diferentes se o significado estiver correto. - Ignora maiúsculas, minúsculas, acentos e pequenos erros de escrita. - Não aceites respostas vagas demais. - Responde apenas em JSON válido. - Não uses markdown. Pergunta: ${pergunta} Resposta esperada: ${respostaCerta} Resposta do aluno: ${respostaAluno} Responde exatamente neste formato: { "certa": true, "feedback": "frase curta" } Ou: { "certa": false, "feedback": "frase curta" } `; const geminiUrl = "https://generativelanguage.googleapis.com/v1beta/models/gemini-1.5-flash:generateContent?key=" + env.GEMINI_API_KEY; const respostaGemini = await fetch(geminiUrl, { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ contents: [ { parts: [ { text: prompt } ] } ], generationConfig: { temperature: 0, responseMimeType: "application/json" } }) }); const dadosGemini = await respostaGemini.json(); if (!respostaGemini.ok) { return Response.json( { certa: false, feedback: "Erro da Gemini: verifica se a chave API está correta." }, { status: 500, headers: corsHeaders } ); } const texto = dadosGemini.candidates?.[0]?.content?.parts?.[0]?.text || ""; let resultado; try { resultado = JSON.parse(texto); } catch { resultado = { certa: false, feedback: "A IA respondeu num formato inválido." }; } if (typeof resultado.certa !== "boolean") { resultado.certa = false; } if (!resultado.feedback) { resultado.feedback = resultado.certa ?
