  <header>
    <div class="caixa">
      <h1>O Meu Quiz</h1>
      <p>Escolhe a resposta certa. Cada pergunta tem 4 opções.</p>
      <p><strong>Pontuação:</strong> <span id="pontuacao">0 certas em 0 respondidas</span></p>
      <button class="recomecar" onclick="recomecarQuiz()">Recomeçar quiz</button>

  <main id="quiz"></main>

  <script>
    const perguntas = [
      {
        pergunta: "Qual é a capital de Portugal?",
        respostas: ["Porto", "Lisboa", "Coimbra", "Braga"],
        certa: 1,
        explicacao: "A capital de Portugal é Lisboa."
      },
      {
        pergunta: "Quanto é 5 + 3?",
        respostas: ["6", "7", "8", "9"],
        certa: 2,
        explicacao: "5 + 3 é igual a 8."
      },
      {
        pergunta: "Qual destes é usado para criar páginas web?",
        respostas: ["HTML", "Excel", "Bluetooth", "Paint"],
        certa: 0,
        explicacao: "HTML é usado para criar a estrutura das páginas web."
      }
    ];

    let respondidas = {};

    function desenharQuiz() {
      const quiz = document.getElementById("quiz");
      quiz.innerHTML = "";

      perguntas.forEach((item, indice) => {
        const caixa = document.createElement("div");
        caixa.className = "caixa";

        let html = `<h2>${indice + 1}. ${item.pergunta}</h2>`;

        item.respostas.forEach((resposta, i) => {
          let classe = "";

          if (respondidas[indice] !== undefined) {
            if (i === item.certa) {
              classe = "certa";
            } else if (i === respondidas[indice]) {
              classe = "errada";
            }
          }

          html += `
            <button class="${classe}" onclick="responder(${indice}, ${i})">
              ${String.fromCharCode(65 + i)}. ${resposta}
            </button>
          `;
        });

        if (respondidas[indice] !== undefined) {
          const acertou = respondidas[indice] === item.certa;

          html += `
            <div class="feedback ${acertou ? "bom" : "mau"}">
              ${acertou ? "Certo!" : "Errado."}
              <br>
              ${item.explicacao}
            </div>
          `;
        }

        caixa.innerHTML = html;
        quiz.appendChild(caixa);
      });

      atualizarPontuacao();
    }

    function responder(numeroPergunta, respostaEscolhida) {
      if (respondidas[numeroPergunta] !== undefined) {
        return;
      }

      respondidas[numeroPergunta] = respostaEscolhida;
      desenharQuiz();
    }

    function atualizarPontuacao() {
      const totalRespondidas = Object.keys(respondidas).length;
      let certas = 0;

      for (let numero in respondidas) {
        if (respondidas[numero] === perguntas[numero].certa) {
          certas++;
        }
      }

      document.getElementById("pontuacao").textContent =
        `${certas} certas em ${totalRespondidas} respondidas`;
    }

    function recomecarQuiz() {
      respondidas = {};
      desenharQuiz();
    }

    desenharQuiz();
  </script>
